# Makefile para gestiÃ³n de templates de proyectos
.PHONY: help init update validate clean install test lint format

# Variables
SCRIPTS_DIR := scripts
TEMPLATES_DIR := .

# Colores para output
GREEN := \033[0;32m
YELLOW := \033[1;33m
RED := \033[0;31m

# Ayuda por defecto
help: ## Mostrar esta ayuda
	@echo "ðŸš€ GestiÃ³n de Templates de Proyectos"
	@echo ""
	@echo "Uso: make [target] [variables]"
	@echo ""
	@echo "Targets disponibles:"

	@echo ""
	@echo "Variables:"
	@echo "  PROJECT_NAME   Nombre del proyecto (default: mi-proyecto)"
	@echo "  PROJECT_TYPE   Tipo de proyecto (default: python-basic)"
	@echo ""
	@echo "Ejemplos:"
	@echo "  make init PROJECT_NAME=mi-api PROJECT_TYPE=nodejs-basic"
	@echo "  make validate PROJECT_NAME=mi-proyecto"

init: ## Inicializar nuevo proyecto
	@echo "$(GREEN)ðŸš€ Inicializando proyecto $(PROJECT_NAME) de tipo $(PROJECT_TYPE)$(NC)"
	@chmod +x $(SCRIPTS_DIR)/init-project.sh
	@$(SCRIPTS_DIR)/init-project.sh $(PROJECT_TYPE) $(PROJECT_NAME) --git --install

init-simple: ## Inicializar proyecto sin Git ni dependencias
	@echo "$(GREEN)ðŸš€ Inicializando proyecto simple $(PROJECT_NAME)$(NC)"
	@chmod +x $(SCRIPTS_DIR)/init-project.sh
	@$(SCRIPTS_DIR)/init-project.sh $(PROJECT_TYPE) $(PROJECT_NAME)

update: ## Actualizar templates desde repositorio
	@echo "$(GREEN)ðŸ”„ Actualizando templates$(NC)"
	@chmod +x $(SCRIPTS_DIR)/update-templates.sh
	@$(SCRIPTS_DIR)/update-templates.sh

validate: ## Validar proyecto generado
	@echo "$(GREEN)ðŸ” Validando proyecto $(PROJECT_NAME)$(NC)"
	@chmod +x $(SCRIPTS_DIR)/validate-project.sh
	@$(SCRIPTS_DIR)/validate-project.sh $(PROJECT_NAME)

clean: ## Limpiar archivos temporales
	@echo "$(YELLOW)ðŸ§¹ Limpiando archivos temporales$(NC)"
	@find . -name "*.pyc" -delete
	@find . -name "__pycache__" -type d -exec rm -rf {} + 2>/dev/null || true
	@find . -name "node_modules" -type d -exec rm -rf {} + 2>/dev/null || true
	@find . -name ".DS_Store" -delete 2>/dev/null || true
	@find . -name "*.log" -delete 2>/dev/null || true
	@echo "$(GREEN)âœ… Limpieza completada$(NC)"

install: ## Instalar herramientas necesarias
	@echo "$(GREEN)ðŸ“¦ Instalando herramientas$(NC)"
	@if command -v brew >/dev/null 2>&1; then \
		echo "Instalando con Homebrew..."; \
		brew install kubectl docker git node python3; \
	elif command -v apt-get >/dev/null 2>&1; then \
		echo "Instalando con apt..."; \
		sudo apt-get update && sudo apt-get install -y kubectl docker.io git nodejs npm python3 python3-pip; \
	elif command -v yum >/dev/null 2>&1; then \
		echo "Instalando con yum..."; \
		sudo yum install -y kubectl docker git nodejs npm python3 python3-pip; \
	else \
		echo "$(RED)âŒ Gestor de paquetes no soportado$(NC)"; \
	fi

test: ## Ejecutar tests de los templates
	@echo "$(GREEN)ðŸ§ª Ejecutando tests$(NC)"
	@for template in python-basic nodejs-basic react-basic; do \
		echo "Probando template: $$template"; \
		rm -rf test-$$template; \
		$(SCRIPTS_DIR)/init-project.sh $$template test-$$template; \
		$(SCRIPTS_DIR)/validate-project.sh test-$$template; \
		rm -rf test-$$template; \
	done
	@echo "$(GREEN)âœ… Tests completados$(NC)"

lint: ## Verificar sintaxis de archivos
	@echo "$(GREEN)ðŸ” Verificando sintaxis$(NC)"
	@echo "Verificando scripts bash..."
	@find $(SCRIPTS_DIR) -name "*.sh" -exec shellcheck {} \; || echo "$(YELLOW)âš ï¸  shellcheck no disponible$(NC)"
	@echo "Verificando archivos YAML..."
	@find . -name "*.yml" -o -name "*.yaml" | head -10 | xargs -I {} sh -c 'python3 -c "import yaml; yaml.safe_load(open(\"{}\"))" && echo "âœ… {}" || echo "âŒ {}"' 2>/dev/null || echo "$(YELLOW)âš ï¸  PyYAML no disponible$(NC)"
	@echo "Verificando archivos JSON..."
	@find . -name "*.json" | head -10 | xargs -I {} sh -c 'python3 -c "import json; json.load(open(\"{}\"))" && echo "âœ… {}" || echo "âŒ {}"' 2>/dev/null || true

format: ## Formatear archivos de cÃ³digo
	@echo "$(GREEN)ðŸŽ¨ Formateando cÃ³digo$(NC)"
	@if command -v prettier >/dev/null 2>&1; then \
		echo "Formateando con Prettier..."; \
		find . -name "*.json" -o -name "*.yml" -o -name "*.yaml" | head -20 | xargs prettier --write; \
	else \
		echo "$(YELLOW)âš ï¸  Prettier no disponible$(NC)"; \
	fi
	@if command -v black >/dev/null 2>&1; then \
		echo "Formateando Python con Black..."; \
		find . -name "*.py" | head -20 | xargs black --line-length 88; \
	else \
		echo "$(YELLOW)âš ï¸  Black no disponible$(NC)"; \
	fi

list: ## Listar templates disponibles
	@echo "$(GREEN)ðŸ“‹ Templates disponibles:$(NC)"
	@ls -1 $(TEMPLATES_DIR) | grep -v scripts | grep -v Makefile | sed 's/^/  - /'

info: ## Mostrar informaciÃ³n del sistema
	@echo "$(GREEN)â„¹ï¸  InformaciÃ³n del sistema:$(NC)"
	@echo "OS: $$(uname -s)"
	@echo "Arquitectura: $$(uname -m)"
	@echo "Shell: $$SHELL"
	@echo ""
	@echo "$(GREEN)ðŸ› ï¸  Herramientas instaladas:$(NC)"
	@command -v git >/dev/null 2>&1 && echo "âœ… Git: $$(git --version)" || echo "âŒ Git no instalado"
	@command -v docker >/dev/null 2>&1 && echo "âœ… Docker: $$(docker --version)" || echo "âŒ Docker no instalado"
	@command -v kubectl >/dev/null 2>&1 && echo "âœ… Kubectl: $$(kubectl version --client --short 2>/dev/null)" || echo "âŒ Kubectl no instalado"
	@command -v node >/dev/null 2>&1 && echo "âœ… Node.js: $$(node --version)" || echo "âŒ Node.js no instalado"
	@command -v python3 >/dev/null 2>&1 && echo "âœ… Python: $$(python3 --version)" || echo "âŒ Python no instalado"
	@command -v java >/dev/null 2>&1 && echo "âœ… Java: $$(java -version 2>&1 | head -n1)" || echo "âŒ Java no instalado"

backup: ## Crear backup de templates
	@echo "$(GREEN)ðŸ’¾ Creando backup$(NC)"
	@mkdir -p backups
	@tar -czf backups/templates-backup-$$(date +%Y%m%d-%H%M%S).tar.gz \
		--exclude='backups' \
		--exclude='.git' \
		--exclude='node_modules' \
		--exclude='__pycache__' \
		.
	@echo "$(GREEN)âœ… Backup creado en backups/$(NC)"

restore: ## Restaurar desde backup (usar BACKUP_FILE=archivo.tar.gz)
	@if [ -z "$(BACKUP_FILE)" ]; then \
		echo "$(RED)âŒ Especifica BACKUP_FILE=archivo.tar.gz$(NC)"; \
		exit 1; \
	fi
	@echo "$(YELLOW)âš ï¸  Restaurando desde $(BACKUP_FILE)$(NC)"
	@tar -xzf $(BACKUP_FILE)
	@echo "$(GREEN)âœ… RestauraciÃ³n completada$(NC)"

stats: ## Mostrar estadÃ­sticas de templates
	@echo "$(GREEN)ðŸ“Š EstadÃ­sticas de templates:$(NC)"
	@echo "Total de templates: $$(ls -1 $(TEMPLATES_DIR) | grep -v scripts | grep -v Makefile | wc -l)"
	@echo "Archivos por template:"
	@for dir in $$(ls -1 $(TEMPLATES_DIR) | grep -v scripts | grep -v Makefile); do \
		if [ -d "$$dir" ]; then \
			count=$$(find "$$dir" -type f | wc -l); \
			echo "  $$dir: $$count archivos"; \
		fi \
	done
	@echo ""
	@echo "Tipos de archivo mÃ¡s comunes:"
	@find . -name "*.py" -o -name "*.js" -o -name "*.json" -o -name "*.yml" -o -name "*.yaml" -o -name "*.md" | \
		sed 's/.*\.//' | sort | uniq -c | sort -nr | head -10

check-deps: ## Verificar dependencias del sistema
	@echo "$(GREEN)ðŸ” Verificando dependencias$(NC)"
	@deps="git docker kubectl node python3 java gradle"; \
	missing=""; \
	for dep in $$deps; do \
		if ! command -v $$dep >/dev/null 2>&1; then \
			missing="$$missing $$dep"; \
		fi \
	done; \
	if [ -n "$$missing" ]; then \
		echo "$(RED)âŒ Dependencias faltantes:$$missing$(NC)"; \
		echo "$(YELLOW)ðŸ’¡ Ejecuta 'make install' para instalarlas$(NC)"; \
	else \
		echo "$(GREEN)âœ… Todas las dependencias estÃ¡n instaladas$(NC)"; \
	fi

dev-setup: ## Configurar entorno de desarrollo
	@echo "$(GREEN)ðŸ› ï¸  Configurando entorno de desarrollo$(NC)"
	@chmod +x $(SCRIPTS_DIR)/*.sh
	@if [ ! -f .git/hooks/pre-commit ]; then \
		echo "Instalando pre-commit hook..."; \
		echo '#!/bin/bash\nmake lint' > .git/hooks/pre-commit; \
		chmod +x .git/hooks/pre-commit; \
	fi
	@echo "$(GREEN)âœ… Entorno configurado$(NC)"
